  // Utility to format seconds as HH:MM:SS

  function formatTime(seconds) {

    const h = Math.floor(seconds / 3600).toString().padStart(2,'0');

    const m = Math.floor((seconds % 3600)/60).toString().padStart(2,'0');

    const s = (seconds % 60).toString().padStart(2,'0');

    return \`\${h}:\${m}:\${s}\`;

  }


  // Data Model in localStorage keys

  const STORAGE_KEY = "life_os_data";


  // Default data structure

  let data = {

    goals: [

      { id: "techWoman", title: "Tech Woman", targetHours: 1000, sessions: [] },

      { id: "myDay", title: "My Day", sessions: [] },

      { id: "habits", title: "Habits Tracker", sessions: [], habits: [

        { id: "earlyRise", title: "Get Up Early", count: 0, streak: 0 },

        { id: "morningRoutine", title: "Morning Routine", count: 0, streak: 0 }

      ] }

    ],

    habits: [

      { id: "earlyRise", title: "Get Up Early", count: 0, streak: 0 },

      { id: "morningRoutine", title: "Morning Routine", count: 0, streak: 0 }

    ],

    dailyReviews: {}

  };


  // Load data from localStorage if exists

  function loadData() {

    const saved = localStorage.getItem(STORAGE_KEY);

    if (saved) {

      try {

        data = JSON.parse(saved);

      } catch(e) {

        console.error("Failed to parse saved data", e);

      }

    }

  }

  // Save data to localStorage

  function saveData() {

    localStorage.setItem(STORAGE_KEY, JSON.stringify(data));

  }


  // Render Goals on page

  function renderGoals() {

    const container = document.getElementById("goals-container");

    container.innerHTML = "";

    data.goals.forEach(goal => {

      // Only render the "techWoman" goal here for session tracking

      if(goal.id === "techWoman") {

        const goalDiv = document.createElement("div");

        goalDiv.className = "goal-block";

        goalDiv.dataset.goalId = goal.id;

        const totalSeconds = goal.sessions.reduce((sum, s) => sum + s.duration, 0);

        const hoursDone = (totalSeconds / 3600).toFixed(1);

        goalDiv.innerHTML = \`

          <div class="goal-header">

            <span class="goal-title">\${goal.title}</span>

            <span class="progress">\${hoursDone}h / \${goal.targetHours}h</span>

          </div>

          <div class="session-list"></div>

        \`;

        container.appendChild(goalDiv);


        // Add click event to toggle session list and timer controls

        goalDiv.onclick = () => {

          if(timerControls.style.display === "none" || currentGoal !== goal.id) {

            openGoal(goal.id);

          } else {

            closeGoal();

          }

        };


        // Render sessions under this goal

        const sessionList = goalDiv.querySelector(".session-list");

        sessionList.innerHTML = "";

        goal.sessions.forEach((session, idx) => {

          const sessionDiv = document.createElement("div");

          sessionDiv.className = "session-item";

          sessionDiv.innerHTML = \`

            <div><strong>Session \${idx + 1}</strong>: <span class="session-time">\${formatTime(session.duration)}</span></div>

          \`;

          sessionList.appendChild(sessionDiv);

          // Clicking session opens notes editor

          sessionDiv.onclick = (ev) => {

            ev.stopPropagation();

            openSessionNotes(goal.id, idx);

          };

        });

      }

    });

  }


  let currentGoal = null;

  const timerControls = document.getElementById("timer-controls");

  const timerDisplay = document.getElementById("timer-display");

  const startBtn = document.getElementById("start-timer-btn");

  const stopBtn = document.getElementById("stop-timer-btn");

  const notesArea = document.getElementById("session-notes");


  let timerInterval = null;

  let sessionStart = null;

  let sessionDuration = 0;


  function openGoal(goalId) {

    currentGoal = goalId;

    timerControls.style.display = "block";

    renderGoals(); // refresh session list highlight

  }

  function closeGoal() {

    currentGoal = null;

    timerControls.style.display = "none";

    clearInterval(timerInterval);

    timerInterval = null;

    sessionStart = null;

    sessionDuration = 0;

    timerDisplay.textContent = "00:00:00";

    notesArea.value = "";

  }


  // Timer functions

  startBtn.onclick = () => {

    if (!currentGoal) return;

    sessionStart = new Date();

    sessionDuration = 0;

    timerDisplay.textContent = "00:00:00";

    startBtn.disabled = true;

    stopBtn.disabled = false;

    notesArea.value = "";

    timerInterval = setInterval(() => {

      sessionDuration = Math.floor((new Date() - sessionStart) / 1000);

      timerDisplay.textContent = formatTime(sessionDuration);

    }, 1000);

  };


  stopBtn.onclick = () => {

    if (!currentGoal || sessionStart === null) return;

    clearInterval(timerInterval);

    timerInterval = null;

    const notes = notesArea.value.trim();

    const goal = data.goals.find(g => g.id === currentGoal);

    if (goal) {

      goal.sessions.push({

        startTime: sessionStart.toISOString(),

        duration: sessionDuration,

        notes: notes

      });

      saveData();

      renderGoals();

      closeGoal();

    }

  };


  // Session notes editor (simple prompt)

  function openSessionNotes(goalId, sessionIdx) {

    const goal = data.goals.find(g => g.id === goalId);

    if (!goal) return;

    const session = goal.sessions[sessionIdx];

    if (!session) return;

    const newNotes = prompt("Edit session notes:", session.notes || "");

    if (newNotes !== null) {

      session.notes = newNotes;

      saveData();

      renderGoals();

    }

  }


  // HABITS SECTION

  const habitsContainer = document.getElementById("habits-container");

  function renderHabits() {

    habitsContainer.innerHTML = "";

    data.habits.forEach(habit => {

      const habitDiv = document.createElement("div");

      habitDiv.className = "habit-block";

      habitDiv.innerHTML = \`

        <div class="habit-header">

          <span class="habit-title">\${habit.title}</span>

          <span class="habit-progress">\${habit.count} / 10000</span>

        </div>

        <div class="habit-streak">Streak: \${habit.streak} days</div>

        <div class="habit-actions">

          <button data-habit-id="\${habit.id}" class="habit-increment">+1</button>

          <button data-habit-id="\${habit.id}" class="habit-reset">Reset</button>

        </div>

      \`;

      habitsContainer.appendChild(habitDiv);

    });


    // Bind habit buttons

    document.querySelectorAll(".habit-increment").forEach(btn => {

      btn.onclick = () => {

        const id = btn.dataset.habitId;

        const habit = data.habits.find(h => h.id === id);

        if (habit) {

          habit.count++;

          habit.streak++;

          saveData();

          renderHabits();

        }

      };

    });

    document.querySelectorAll(".habit-reset").forEach(btn => {

      btn.onclick = () => {

        const id = btn.dataset.habitId;

        const habit = data.habits.find(h => h.id === id);

        if (habit) {

          habit.count = 0;

          habit.streak = 0;

          saveData();

          renderHabits();

        }

      };

    });

  }

<script>

  // MY DAY calendar and daily review

  const calendarContainer = document.getElementById("calendar-container");

  const dailyReviewSection = document.getElementById("daily-review");

  const selectedDateSpan = document.getElementById("selected-date");

  const dailyTasksDiv = document.getElementById("daily-tasks");


  function getToday() {

    return new Date();

  }

  let selectedDate = getToday();


  function formatDate(date) {

    return \`\${date.getFullYear()}-\${(date.getMonth()+1).toString().padStart(2,'0')}-\${date.getDate().toString().padStart(2,'0')}\`;

  }


  function renderCalendar(year, month) {

    calendarContainer.innerHTML = "";

    const table = document.createElement("table");

    table.className = "calendar";


    const daysOfWeek = ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"];

    const thead = document.createElement("thead");

    const trHead = document.createElement("tr");

    daysOfWeek.forEach(d => {

      const th = document.createElement("th");

      th.textContent = d;

      trHead.appendChild(th);

    });

    thead.appendChild(trHead);

    table.appendChild(thead);


    const firstDay = new Date(year, month, 1);

    const startDay = firstDay.getDay();

    const daysInMonth = new Date(year, month + 1, 0).getDate();


    const tbody = document.createElement("tbody");

    let row = document.createElement("tr");


    // Empty cells before first day

    for (let i = 0; i < startDay; i++) {

      const td = document.createElement("td");

      row.appendChild(td);

    }


    for (let day = 1; day <= daysInMonth; day++) {

      if ((startDay + day - 1) % 7 === 0 && day !== 1) {

        table.appendChild(row);

        row = document.createElement("tr");

      }

      const td = document.createElement("td");

      td.textContent = day;

      const dateKey = \`\${year}-\${(month + 1).toString().padStart(2,'0')}-\${day.toString().padStart(2,'0')}\`;

      if (dateKey === formatDate(selectedDate)) {

        td.classList.add("selected");

      }

      if (dateKey === formatDate(getToday())) {

        td.classList.add("today");

      }

      td.onclick = () => {

        selectedDate = new Date(year, month, day);

        renderCalendar(year, month);

        showDailyReview(selectedDate);

      };

      row.appendChild(td);

    }


    // Empty cells after last day

    const remainingCells = 7 - ((startDay + daysInMonth) % 7);

    if (remainingCells < 7) {

      for (let i = 0; i < remainingCells; i++) {

        const td = document.createElement("td");

        row.appendChild(td);

      }

    }

    table.appendChild(row);

    calendarContainer.appendChild(table);

  }


  function showDailyReview(date) {

    dailyReviewSection.style.display = "block";

    selectedDateSpan.textContent = formatDate(date);

    dailyTasksDiv.innerHTML = "";


    const key = formatDate(date);


    // List sessions from "Tech Woman" goal on this date

    const techGoal = data.goals.find(g => g.id === "techWoman");

    const sessionsToday = techGoal.sessions.filter(session => {

      return session.startTime.startsWith(key);

    });


    if (sessionsToday.length > 0) {

      const h4 = document.createElement("h4");

      h4.textContent = "Tech Woman Sessions:";

      dailyTasksDiv.appendChild(h4);

      sessionsToday.forEach((session, i) => {

        const div = document.createElement("div");

        div.textContent = \`Session \${i+1}: Duration \${formatTime(session.duration)}; Notes: \${session.notes || "None"}\`;

        dailyTasksDiv.appendChild(div);

      });

    } else {

      dailyTasksDiv.innerHTML = "<p>No sessions recorded for this day.</p>";

    }


    // List habits marked for this day if any

    if (data.dailyReviews[key] && data.dailyReviews[key].habitsDone) {

      const habitsDone = data.dailyReviews[key].habitsDone;

      const h4h = document.createElement("h4");

      h4h.textContent = "Habits done today:";

      dailyTasksDiv.appendChild(h4h);

      habitsDone.forEach(hid => {

        const habit = data.habits.find(h => h.id === hid);

        if (habit) {

          const div = document.createElement("div");

          div.textContent = habit.title;

          dailyTasksDiv.appendChild(div);

        }

      });

    }


    // Button to mark habits done today

    const habitMarkDiv = document.createElement("div");

    habitMarkDiv.style.marginTop = "10px";

    habitMarkDiv.innerHTML = "<h4>Mark Habits Done Today:</h4>";

    data.habits.forEach(habit => {

      const btn = document.createElement("button");

      btn.textContent = habit.title;

      btn.style.marginRight = "6px";

      btn.onclick = () => {

        markHabitDoneToday(habit.id, key);

        showDailyReview(date);

        renderHabits();

      };

      habitMarkDiv.appendChild(btn);

    });

    dailyTasksDiv.appendChild(habitMarkDiv);

  }


  function markHabitDoneToday(habitId, dateKey) {

    if (!data.dailyReviews[dateKey]) {

      data.dailyReviews[dateKey] = { habitsDone: [] };

    }

    if (!data.dailyReviews[dateKey].habitsDone.includes(habitId)) {

      data.dailyReviews[dateKey].habitsDone.push(habitId);

      // Increment habit count & streak

      const habit = data.habits.find(h => h.id === habitId);

      if (habit) {

        habit.count++;

        habit.streak++;

      }

      saveData();

    }

  }


  // EXPORT / IMPORT data backup and restore

  const exportBtn = document.getElementById("export-data-btn");

  const importBtn = document.getElementById("import-data-btn");

  const backupTextarea = document.getElementById("backup-textarea");


  exportBtn.onclick = () => {

    backupTextarea.value = JSON.stringify(data, null, 2);

    alert("Data exported below. You can copy and save it externally.");

  };


  importBtn.onclick = () => {

    try {

      const imported = JSON.parse(backupTextarea.value);

      if (imported && typeof imported === "object") {

        data = imported;

        saveData();

        renderGoals();

        renderHabits();

        renderCalendar(selectedDate.getFullYear(), selectedDate.getMonth());

        showDailyReview(selectedDate);

        alert("Data imported successfully!");

      } else {

        alert("Invalid JSON data.");

      }

    } catch(e) {

      alert("Error parsing JSON: " + e.message);

    }

  };


  // INITIALIZATION

  loadData();

  renderGoals();

  renderHabits();


  // Show calendar and daily review section

  document.getElementById("my-day-calendar").style.display = "block";

  renderCalendar(selectedDate.getFullYear(), selectedDate.getMonth());

  showDailyReview(selectedDate);


</script>
